pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                // Clone the Git repository
                git url: 'https://github.com/harini1810/pipeline.git', branch: 'main' // Update with your repo URL and branch
            }
        }

        stage('Compile') {
            steps {
                script {
                    // Change directory and compile the Java file
                    bat '''
                        javac -d target hello.java
                    '''
                }
            }
        }

        stage('Run') {
            steps {
                script {
                    // Run the compiled Java program
                    bat '''
                        java -cp target hello
                    '''
                }
            }
        }
    }

    post {
        always {
            echo 'Pipeline completed.'
        }
        success {
            echo 'Build succeeded!'
        }
        failure {
            echo 'Build failed.'
        }
    }
}
